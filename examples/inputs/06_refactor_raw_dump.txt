===== src/pricing/format.py =====
def format_price(amount):
    """Return a human-readable price string for the given amount in cents."""
    dollars = amount / 100
    return f"${dollars:,.2f}"

def format_discount(amount, pct):
    base = format_price(amount)
    return f"{base} (-{pct}%)"

===== src/pricing/cart.py =====
from src.pricing.format import format_price

class Cart:
    def __init__(self):
        self.items = []

    def add(self, sku, amount):
        self.items.append((sku, amount))

    def total_label(self):
        total = sum(a for _, a in self.items)
        return "Total: " + format_price(total)

    def line_labels(self):
        return [f"{sku}: {format_price(a)}" for sku, a in self.items]

===== src/pricing/invoice.py =====
from src.pricing.format import format_price

def render_invoice(lines):
    out = []
    for sku, amount in lines:
        out.append(f"{sku}\t{format_price(amount)}")
    out.append("---")
    out.append("TOTAL\t" + format_price(sum(a for _, a in lines)))
    return "\n".join(out)
